    function isFirstElementInTheLine(child,parent) {

        //special bonus for those using jQuery
        if (child instanceof jQuery) {
            child = child[0];
        }

        if (parent instanceof jQuery) {
            parent = parent[0];
        }

        var rect = child.getBoundingClientRect();
        var parentrect = parent.getBoundingClientRect();

        return (
            rect.left == parentrect.left
        );
    }

    var resizeId;
    var oldwidth;
    var stampWidth = 220;
    
    function doneResizing() {
        var nextpage = $(&#39;.next-page-item&#39;);
        var parent = $(&#39;#product-list&#39;);
        oldwidth = $(window).width();
        if (nextpage.length && parent.length )
            if (isFirstElementInTheLine(nextpage, parent)) {
                nextpage.hide();
        } else {
                nextpage.css(&#39;display&#39;, &#39;inline-block&#39;);
        }
    }

    $(document).ready(function() {
        doneResizing();

        $(window).resize(function () {
            if ($(window).width() > (oldwidth + stampWidth) || $(window).width() < (oldwidth - stampWidth)) {
                clearTimeout(resizeId);
                resizeId = setTimeout(doneResizing, 500);
            }
        });
    });
